# Code
```c++
class Solution {
public:
    vector<vector<int>> result;
    vector<vector<int>> permute(vector<int>& nums) {
        vector<bool> visited(nums.size(), 0);
        vector<int> cur;
        dfs(visited, cur, nums);
        return result;
    }

    void dfs(vector<bool> visited, vector<int> cur, vector<int> nums) {
        if (cur.size() == nums.size()) {
            result.push_back(cur);
            return;
        }
        for (int i=0; i<nums.size(); i++) {
            if (!visited[i]) {
                visited[i] = true;
                cur.push_back(nums[i]);
                dfs(visited, cur, nums);
                cur.pop_back();
                visited[i] = false;
            }
        }
    }
};
```
