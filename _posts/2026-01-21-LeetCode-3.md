# Code
```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> exist;
        int left = 0, right = 0, ans = 0;

        while (right < (int)s.size()) {
            if (!exist.count(s[right])) {
                exist.insert(s[right]);
                ans = max(ans, right - left + 1);
                right++;
            } else {
                exist.erase(s[left]);
                left++;
            }
        }
        return ans;
    }
};
```
