    // Sieve of Eratosthenes
    public static int countPrimes(int n) {
        if (n <= 2) return 0;
        boolean[] isPrime = new boolean[n];
        for (int i=3; i<n; i+=2) {
            isPrime[i] = true;
        }
        isPrime[2] = true;
        for (int i=3; i<Math.sqrt(n); i+=2) {
            if (isPrime[i]) { // if i is prime
                for (int j=i*i; j<n; j+=i*2) { // mark i*i, i*(i+1), i*(i+2),... as non-prime
                    isPrime[j] = false;
                }
            }
        }
        int result = 0;
        for (int i=2; i<n; i++) {
            if (isPrime[i]) result++;
        }
        return result;
    }
}
