# Code
```c++
class Solution {
public:
    int maxArea(vector<int>& height) {
        int left = 0, right = height.size() - 1;
        int ma = 0;
        while (left < right) {
            ma = max(ma, (right- left) * min(height[left], height[right]));
            if (height[left] < height[right]) {
                left++;
            }else{
                right--;
            }
        }
        return ma;
    }
};
```
