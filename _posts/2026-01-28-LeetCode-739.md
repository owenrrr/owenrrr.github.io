class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        vector<int> result(t.size(), 0);
        vector<int> stk;

        int ind = 0, tp = 0;

        for (int i=0; i<t.size(); i++) {
            while (!stk.empty()) {
                ind = stk[stk.size()-1];
                tp = t[ind];
                if (tp >= t[i]) {
                    break;
                }
                result[ind] = i - ind;
                stk.pop_back();
            }
            stk.push_back(i);
        }

        return result;
    }
};
