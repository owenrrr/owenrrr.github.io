class Solution {
public:
    int search(vector<int>& nums, int target) {
        if (nums.size() == 0) return -1;

        int base = 0, ceil = nums.size() - 1;
        int pt = (base + ceil) / 2;

        while (base <= ceil) {
            if (nums[pt] == target) {
                return pt;
            }else if (nums[pt] > target) {
                ceil = pt - 1;
            }else {
                base = pt + 1;
            }
            pt = (base + ceil) / 2;
        }
        return -1;
    }
};
